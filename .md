graph TD
    %% --- 专转 注爪 ---
    classDef folder fill:#f0f7ff,stroke:#005b9f,stroke-width:2px,color:#003b6f,rx:5,ry:5;
    classDef pythonFile fill:#fff3e0,stroke:#d84315,stroke-width:1px,color:#bf360c,stroke-dasharray: 3 3;
    classDef configFile fill:#f3e5f5,stroke:#8e24aa,stroke-width:1px,color:#4a148c,shape:can;
    classDef dockerFile fill:#e8f5e9,stroke:#2e7d32,stroke-width:2px,color:#1b5e20,shape:hex;

    subgraph ProjectRoot[ black-coordinates-list Project Root]
        direction TB

        ENV[.env]:::configFile
        README[README.md]:::rootFile

        %% --- Service A Structure ---
        subgraph ServiceA_Dir[ service-a]
            REQ_A[requirements]:::configFile
            
            subgraph AppA_Container[Docker Build Context A]
                direction TB
                DA[Dockerfile]:::dockerFile
                
                subgraph AppA_Dir[ app/]
                    MA[main]:::pythonFile
                    RA[routes]:::pythonFile
                    SA[services]:::pythonFile
                    SCA[schemas]:::pythonFile
                end
            end
        end

        %% --- Service B Structure ---
        subgraph ServiceB_Dir[ service-b]
            REQ_B[requirements.txt]:::configFile
            
            subgraph AppB_Container[Docker Build Context B]
                direction TB
                DB[Dockerfile]:::dockerFile

                subgraph AppB_Dir[ app/]
                    MB[main]:::pythonFile
                    RB[routes]:::pythonFile
                    STB[storage]:::pythonFile
                    SCB[schemas]:::pythonFile
                end
            end
        end

        end

    end

    DB ==>|Builds| AppB_Dir

    %% 拽砖专 转砖转转 专拽住专爪

    DC ==>|Orchestrates Build & Run| DB
    
    DA -.->|Creates Image| RunningA
